## 二、API 服务层封装

### 2.1 请求封装 `services/request.ts`

typescript

```
// services/request.ts
const BASE_URL = 'http://localhost:8000/api'

interface Response<T = any> {
  code: number
  message: string
  data: T
}

// 错误码常量
export const ErrorCode = {
  SUCCESS: 0,
  PARAM_ERROR: 40001,
  USER_NOT_FOUND: 40002,
  RECORD_EXISTS: 40003,
  FEATURE_LOCKED: 40004,
  INVITE_EXISTS: 40005,
  INVITE_SELF: 40006,
  GUARDIAN_EXISTS: 40007,
  COUPLE_EXISTS: 40008,
  UNAUTHORIZED: 40101,
  FORBIDDEN: 40301,
  NOT_FOUND: 40401,
  RATE_LIMIT: 42901,
  SERVER_ERROR: 50001,
}

// 获取存储的 token
const getToken = (): string | null => {
  return uni.getStorageSync('token') || null
}

// 通用请求方法
export const request = <T = any>(
  url: string,
  options: UniApp.RequestOptions = {}
): Promise<Response<T>> => {
  return new Promise((resolve, reject) => {
    const token = getToken()
    const header: Record<string, string> = {
      'Content-Type': 'application/json',
      ...options.header,
    }
    if (token) {
      header['Authorization'] = `Bearer ${token}`
    }

    uni.request({
      url: BASE_URL + url,
      method: options.method || 'GET',
      data: options.data,
      header,
      success: (res) => {
        const response = res.data as Response<T>
        if (response.code === ErrorCode.UNAUTHORIZED) {
          // token 过期，清除登录状态并跳转登录
          uni.removeStorageSync('token')
          uni.removeStorageSync('userInfo')
          uni.reLaunch({ url: '/pages/index/index' })
          reject(new Error('登录已过期'))
        } else if (response.code !== ErrorCode.SUCCESS) {
          // 其他业务错误
          uni.showToast({ title: response.message, icon: 'none' })
          reject(new Error(response.message))
        } else {
          resolve(response)
        }
      },
      fail: (err) => {
        uni.showToast({ title: '网络错误', icon: 'none' })
        reject(err)
      },
    })
  })
}

// GET 请求
export const get = <T = any>(url: string, params?: Record<string, any>): Promise<Response<T>> => {
  let queryString = ''
  if (params) {
    const searchParams = new URLSearchParams()
    Object.entries(params).forEach(([key, value]) => {
      if (value !== undefined && value !== null) {
        searchParams.append(key, String(value))
      }
    })
    queryString = searchParams.toString()
  }
  const fullUrl = queryString ? `${url}?${queryString}` : url
  return request<T>(fullUrl, { method: 'GET' })
}

// POST 请求
export const post = <T = any>(url: string, data?: any): Promise<Response<T>> => {
  return request<T>(url, { method: 'POST', data })
}
```

## 三、Pinia 状态管理

### 3.1 用户 Store `stores/user.ts`

typescript

```
// stores/user.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { userApi, type UserInfo } from '@/services/user'
import { inviteApi } from '@/services/invite'

export const useUserStore = defineStore('user', () => {
  // State
  const userInfo = ref<UserInfo | null>(null)
  const token = ref<string>('')
  const isLoggedIn = ref(false)

  // Getters
  const inviteCount = computed(() => userInfo.value?.invite_count || 0)
  const unlockedFeatures = computed(() => userInfo.value?.unlocked_features || [])
  const isFeatureUnlocked = (feature: string) => unlockedFeatures.value.includes(feature)

  // Actions
  const login = async (code: string, inviterId?: string) => {
    const res = await userApi.login({ code, inviter_id: inviterId })
    if (res.code === 0) {
      token.value = res.data.token
      userInfo.value = {
        user_id: res.data.user_id,
        nickname: '用户',
        avatar: '👤',
        invite_count: res.data.invite_count,
        continuous_days: 0,
        unlocked_features: res.data.unlocked_features,
        badges: [],
      }
      isLoggedIn.value = true
      // 存储到本地
      uni.setStorageSync('token', res.data.token)
      uni.setStorageSync('userInfo', userInfo.value)
      return res.data
    }
    throw new Error(res.message)
  }

  const fetchUserInfo = async () => {
    const res = await userApi.getUserInfo()
    if (res.code === 0) {
      userInfo.value = res.data
      uni.setStorageSync('userInfo', userInfo.value)
    }
    return userInfo.value
  }

  const logout = () => {
    token.value = ''
    userInfo.value = null
    isLoggedIn.value = false
    uni.removeStorageSync('token')
    uni.removeStorageSync('userInfo')
  }

  const initFromStorage = () => {
    const storedToken = uni.getStorageSync('token')
    const storedUserInfo = uni.getStorageSync('userInfo')
    if (storedToken && storedUserInfo) {
      token.value = storedToken
      userInfo.value = storedUserInfo
      isLoggedIn.value = true
    }
  }

  return {
    userInfo,
    token,
    isLoggedIn,
    inviteCount,
    unlockedFeatures,
    isFeatureUnlocked,
    login,
    fetchUserInfo,
    logout,
    initFromStorage,
  }
})
```



### 3.2 记录 Store `stores/record.ts`

typescript

```
// stores/record.ts
import { defineStore } from 'pinia'
import { ref } from 'vue'
import { recordApi, type CalendarData, type TodayStatus } from '@/services/record'
import { useUserStore } from './user'

export const useRecordStore = defineStore('record', () => {
  const userStore = useUserStore()
  
  const calendarData = ref<CalendarData | null>(null)
  const todayStatus = ref<TodayStatus | null>(null)
  const continuousDays = ref(0)

  const fetchCalendar = async (year: number, month: number) => {
    const res = await recordApi.getCalendar({ year, month })
    if (res.code === 0) {
      calendarData.value = res.data
    }
    return calendarData.value
  }

  const fetchTodayStatus = async () => {
    const res = await recordApi.getTodayStatus()
    if (res.code === 0) {
      todayStatus.value = res.data
    }
    return todayStatus.value
  }

  const createRecord = async (recordType: string, recordDate: string, recordValue?: any) => {
    const res = await recordApi.create({
      record_type: recordType,
      record_date: recordDate,
      record_value: recordValue,
    })
    if (res.code === 0) {
      continuousDays.value = res.data.continuous_days
      // 刷新相关数据
      await fetchTodayStatus()
      const now = new Date()
      await fetchCalendar(now.getFullYear(), now.getMonth() + 1)
      return res.data
    }
    throw new Error(res.message)
  }

  const checkCanRecord = (recordType: string): boolean => {
    return userStore.isFeatureUnlocked(recordType)
  }

  return {
    calendarData,
    todayStatus,
    continuousDays,
    fetchCalendar,
    fetchTodayStatus,
    createRecord,
    checkCanRecord,
  }
})
```

const currentMonth = ref(5)
const isExpanded = ref(false)
const lockPopupVisible = ref(false)
const lockTarget = ref(1)

// 计算属性
const inviteCount = computed(() => userStore.inviteCount)
const isSleepUnlocked = computed(() => userStore.isFeatureUnlocked('sleep'))
const todayStatus = computed(() => recordStore.todayStatus)

// 固定记录类型
const visibleRecordTypes = ref([
  { id: 'poop', icon: '💩', name: '拉屎', bgColor: '#e8f8ef', unlocked: true },
  { id: 'period', icon: '🩸', name: '姨妈', bgColor: '#ffebee', unlocked: true },
])

// 未解锁记录类型（示例）
const lockedRecordTypes = ref([
  { id: 'sleep', icon: '😴', name: '睡眠', needInvite: 1 },
  { id: 'water', icon: '💧', name: '喝水', needInvite: 3 },
  { id: 'anime', icon: '📺', name: '追番', needInvite: 8 },
  { id: 'sport', icon: '🏋️', name: '运动', needInvite: 15 },
])

// 日历数据
const weekdays = ['日', '一', '二', '三', '四', '五', '六']
const calendarDays = ref<Array<{ day: number; isToday: boolean; hasRecord: boolean }>>([])

// 方法
const loadCalendar = async () => {
  const data = await recordStore.fetchCalendar(currentYear.value, currentMonth.value)
  if (data) {
    calendarDays.value = data.days.map(d => ({
      day: d.day,
      isToday: d.is_today || false,
      hasRecord: d.has_record,
    }))
  }
}

const loadTodayStatus = async () => {
  await recordStore.fetchTodayStatus()
}

const prevMonth = () => {
  if (currentMonth.value === 1) {
    currentMonth.value = 12
    currentYear.value--
  } else {
    currentMonth.value--
  }
  loadCalendar()
}

const nextMonth = () => {
  if (currentMonth.value === 12) {
    currentMonth.value = 1
    currentYear.value++
  } else {
    currentMonth.value++
  }
  loadCalendar()
}

const onRecordClick = async (item: any) => {
  if (!item.unlocked) {
    lockTarget.value = item.needInvite
    lockPopupVisible.value = true
    return
  }
  
  try {
    const today = dayjs().format('YYYY-MM-DD')
    await recordStore.createRecord(item.id, today)
    uni.showToast({ title: '记录成功', icon: 'success' })
  } catch (error) {
    // 错误已在 request 中处理
  }
}

const onLockedClick = (item: any) => {
  lockTarget.value = item.needInvite
  lockPopupVisible.value = true
}

const toggleExpand = () => {
  isExpanded.value = !isExpanded.value
}

const goToInvite = () => {
  uni.navigateTo({ url: '/pages/invite/index' })
}

const onDayClick = (day: any) => {
  // 可选：弹出该日详情
  console.log('点击日期', day)
}

const initLogin = async () => {

onMounted(() => {
  userStore.initFromStorage()
  if (!userStore.isLoggedIn) {
    initLogin()
  } else {
    loadTodayStatus()
    loadCalendar()
  }
})
